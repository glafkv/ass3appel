How to run:	
	make 		//to compile
	./oss -h	//to show the help menu
	./oss -s x	//to set a max number of child processes spawned. the default is 5
	./oss -l <filename>	//the log file used
	./oss -t z		//the time in real seconds when the master will termindate itself and all children
	make clean	//to erase all object and executable files
